Вспомним прошлый материал

Вспомним прошлое

        
          let name = "Petro";

          function sayMyName() {
              console.log(name);
          }
          
          sayMyName();

Что выведет код?

Ответ

        
          Petro

Вернемся к настоящему

В нашей программе часто нужно вычислять среднее значение массива чисел.

        
          let numbers = [12, 14, 14];

          let sum = 0;
          let count = 0;
          
          for (let index = 0; index < numbers.length; index++) {
            count += 1;
            sum += numbers[index];
          }
          
          console.log(sum / count);

Есть проблема

Не хорошо, когда мы повторяем такой простой код сотни раз.

Мы можем вынести этот код в функцию! Но нужно как-то передать ему массив с числами.

Для этого используются аргументы функций. Это такой способ явно передать данные в функцию.

Решение

В нашей программе часто нужно вычислять среднее значение массива чисел.

        
          function average(numbers) {
            let sum = 0;
            let count = 0;
            for (let index = 0; index < numbers.length; index++) {
                count += 1;
                sum += numbers[index];
            }
        
            console.log(sum / count);
          }
          
          average([12, 14, 14]);
          
          let array = [1, 2, 3, 4, 5];
          average(array);

Что выведет этот код?

Ответ

То же самое!

        
          13,333333333

Мы передали массив чисел как аргумент.

Аргументы

Аргументов может быть сколько угодно!

        
          function sayHello(firstName, lastName) {
            console.log('Hello, ' + firstName + ' ' + lastName);
          }
          
          sayHello('Petro', 'Egorov');

Что есть не передать аргументы?

        
          function sayHello(firstName, lastName) {
            console.log('Hello, ' + firstName + ' ' + lastName);
          }
          
          sayHello('Petro');

Что выведет этот код?

Ответ

        
          Hello, Petro undefined

Если не передать аргумент, ему будет присвоено значение undefined.

Значение по умолчанию

        
          function sayHello(firstName, lastName = 'Surname') {
            console.log('Hello, ' + firstName + ' ' + lastName);
          }
          
          sayHello('Petro');

Hello, Petro Surname

Вопрос!

Какие применения для функций с аргументами вы можете найти прямо сейчас?

Мы хотим от функции большей универсальности!

После вычисления среднего значения, его не всегда нужно выводить в консоль. Иногда хочется записать в файл, иногда вывести в alert.

Что делать?

Наивное решение

        
          function averageInConsole(numbers) {
              // ...
              console.log(sum / count);
          }
          
          function averageInAlert(numbers) {
              // ...
              alert(sum / count);
          }

Копирование кода…

Для этого случая подходит возвращаемое значение.

Функция может вернуть какое-то значение тому коду, который ее вызвал.

        
          function average(numbers) {
            let sum = 0;
            let count = 0;
            for (let index = 0; index < numbers.length; index++) {
                count += 1;
                sum += numbers[index];
            }
        
            return sum / count;
          }
          
          let avg = average([12, 14, 14]);
          console.log(avg);

Это обычное значение!

С возвращенным значением мы можем делать уже что угодно.

        
          let avgForConsole = average([12, 14, 14]);
          console.log(avgForConsole);
          
          let avgForAlert = average([120, 1, 543]);
          alert(avgForAlert);
          
          let alertForFile = average([54, 32, 12]);
          writeFile(alertForFile);

Завершение

return завершает работу функции.

        
          function average(numbers) {
            let sum = 0;
            let count = 0;
            for (let index = 0; index < numbers.length; index++) {
                count += 1;
                sum += numbers[index];
            }
            
            if (count === 0) {
                return;
            }
        
            return sum / count;
          }

Как только встретился return — функция прекращает выполняться.

Вопрос

Сколько значений можно вернуть из функции?

Ответ

Технически — одно.

        
          function average(numbers) {
            // ...
        
            return sum / count;
          }

Но хочется несколько…

Тогда можно вернуть массив или объект.

        
          function average(numbers) {
            // ...
        
            let avg = sum / count;
            
            return { avg: avg, sum: sum, total: count };
          }
